QUESTION 01:
SELECT EMP_NAME, JOB, SALARY
FROM EMP
WHERE JOB='Salesman' AND EMP_NAME LIKE'%E%';





QUESTION 02:
SELECT ENAME, SALARY, HIREDATE, SALARY *12 AS Annual_Sal, JOB
FROM EMP
WHERE (JOB='MANAGER' OR JOB='SALESMAN')
AND (ENAME LIKE '_A%' OR ENAME LIKE '__A%') 
AND SALARY BETWEEN 1000 AND 2500
AND TO_CHAR(HIREDATE, 'MON') NOT IN('FEB', 'MAY')
ORDER BY Annual_Sal DESC;





QUESTION 03:
SELECT ENAME, JOB, HIREDATE, SAL, DEPTNO
FROM EMP
WHERE MGR <> 7839
AND COMM IS NOT NULL
AND HIREDATE BETWEEN TO_DATE('21-Feb-81', 'DD-Mon-YY') AND TO_DATE('17-Dec-81' , 'DD-Mon-YY')
ORDER BY SAL DESC, COMM DESC;





QUESTION 04:
SELECT ENAME AS 'Name',INSTR(ENAME, 'A' 1,2) AS 'Position',SAL AS 'Salary'
FROM EMP
WHERE LENGTH(ENAME) - LENGTH(REPLACE(ENAME,'A', '')) = 2;





QUESTION 05:
SELECT CONCAT(UPPER(SUBSTRING(ENAME,1 ,1)),LOWER(SUBSTRING(ENAME, 2)))AS "Formatted Name"
FROM EMP 
WHERE ENAME LIKE 'A%' OR ENAME LIKE "E%" OR ENAME LIKE "B%"
ORDER BY "ENAME";



QUESTION 06:
SELECT DEPTNO AS 'Department', SUMM(SAL) AS 'TotalSalary'
FROM EMP
GROUP BY DEPTNO
ORDER NY TotalSalary ASC;




QUESTION 07:
SELECT ENAME AS'Name', DEPTNO AS 'Department', SAL * 1.15 AS 'Incremented_Salary'
FROM EMP
WHERE (JOB='Clerk' OR JOB='Salesman')
AND SAL NOT IN (1250,1300,1400)
AND ENAME LIKE '%A%';




QUESTION 08:
SELECT
    e1.ENAME AS 'Employee_Name',
    e1.DEPTNO AS 'Department_Number',
    e2.ENAME AS 'Colleague_Name'
FROM EMP e1
JOIN EMP e2 ON e1.DEPTNO = e2.DEPTNO
           AND e1.ENAME <> e2.ENAME
ORDER BY e1.DEPTNO, e1.ENAME, e2.ENAME;
